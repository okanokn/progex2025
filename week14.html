14.3 グラフ

グラフで表現されるデータについて，頂点\(v_i\)から\(v_j\)への最短経路を調べたり，グラフが閉路を持つか否かを判定したい場合がある． そのような場合に，すべての頂点を辺をたどって確認する手順が必要となる．

以下では無向グラフ\(G(V, E)\)について，深さ優先探索（depth-first search）と幅優先探索（breadth-first search）の2種類の探索方法を解説する．

グラフをプログラム上で表現する方法として，ここでは隣接行列を用いる．隣接行列\(A\in\mathbb{R}^{|V|\times |V|}\)の\((i, j)\)成分\(a_{ij}\)はつぎのように定義される: \[ a_{ij}= \begin{cases} 1 & \text{if } (v_i,v_j)\in E,\\ 0 & \text{otherwise}. \end{cases} \] つまり，\(v_i\)と\(v_j\)が辺で繋がっていれば\(1\)，そうでなければ\(0\)をとる． 無向グラフであれば\(A\)は対称行列である．

無向グラフ Figure 14.1 の隣接行列を，グローバル変数の2次元配列で定義する実装例を示す．

#include <stdio.h>
#define N 5           // 頂点数
int adj[N][N] = {0};  // 隣接行列

int main(void) {
  // グラフの定義
  adj[0][1] = adj[1][0] = 1;
  adj[0][2] = adj[2][0] = 1;
  adj[1][3] = adj[3][1] = 1;
  adj[1][4] = adj[4][1] = 1;
  adj[2][4] = adj[4][2] = 1;
  return 0;
}

14.4 深さ優先探索

深さ優先探索は，辺をたどって深く潜ってから戻って来る探索方法である． 今見ている頂点が複数の未訪問頂点に隣接する場合は，どれかひとつを選んで先に進み，進んだ先の頂点をすべて訪問してから分岐点に戻って来る．

深さ優先探索は再帰を使って実現できる． または，再帰によるコールスタックを明示的にスタックとして実装することでも実現可能である．

以下は再帰を用いた実装例である．

// グローバル変数 各頂点の訪問フラグ
int visited[N] = {0};  // 0: 未訪問, 1: 済み

// 頂点vから深さ優先探索する関数
void dfs(int v) {
  visited[v] = 1;   // 訪問済みに設定して
  printf("%d ", v); // 頂点番号を出力
  for (int i = 0; i < N; i++) {
    if (adj[v][i] && !visited[i]) {  // 隣接していて未訪問なら
      dfs(i);                        // 先に進む
    }
  }
}
main関数を含む実装例 
#include <stdio.h>
#define N 5
int adj[N][N] = {0};   // 隣接行列
int visited[N] = {0};  // 各頂点の訪問フラグ. 0: 未, 1: 済み

void dfs(int v) {
  visited[v] = 1;
  printf("%d ", v);
  for (int i = 0; i < N; i++) {
    if (adj[v][i] && !visited[i]) {
      dfs(i);  // 未訪問の頂点に進む
    }
  }
}

int main(void) {
  // グラフの定義
  adj[0][1] = adj[1][0] = 1;
  adj[0][2] = adj[2][0] = 1;
  adj[1][3] = adj[3][1] = 1;
  adj[1][4] = adj[4][1] = 1;
  adj[2][4] = adj[4][2] = 1;

  dfs(0);

  return 0;
}

深さ優先探索のなかでも，行きがけに頂点をチェックして訪問済みとする行きがけ順と，先が無い頂点から帰りがけに行うか帰りがけ順がある．上の実装は行きがけ順である．

14.5 幅優先探索

幅優先探索は，ある頂点に隣接する頂点をすべて調べてから先の頂点に進む探索方法である．

幅優先探索は，訪問した頂点に隣接する頂点をキューに順次追加して，キューの先頭から順に訪問する方法で実装できる． 以下は，第11週で紹介したキューの実装例を用いた幅優先探索関数の例である．

// 頂点startから幅優先探索を行う関数
void bfs(Queue *q, int start) {
  visited[start] = 1;
  enqueue(q, start);  // 初期頂点をキューに追加
  while (!isEmpty(q)) {
    int v;
    dequeue(q, &v);  // キュー先頭のv
    printf("%d ", v);
    for (int i = 0; i < N; i++) {// vの隣接頂点を順にチェック
      if (adj[v][i] && !visited[i]) {
        visited[i] = 1;  // vに隣接する未訪問頂点
        enqueue(q, i);   // キューに追加して次の隣接頂点へ
      }
    }
  }
}

12行目の，発見した隣接頂点iをキューに追加しつつ，vの隣接頂点の確認を続行するところがポイントである．

main関数を含む実装例 
#include <stdio.h>
#include <stdlib.h>
#define N 5
int adj[N][N] = {0};
int visited[N] = {0};

typedef struct node {
  int data;
  struct node *next;
} Node;

typedef struct {
  Node *front;
  Node *rear;
} Queue;

void initQueue(Queue *q) {
  q->front = NULL;
  q->rear = NULL;
}

int isEmpty(Queue *q) { return q->front == NULL; }

int enqueue(Queue *q, int data) {
  Node *newNode = (Node *)malloc(sizeof(Node));
  if (newNode == NULL) {
    return -1;
  }
  newNode->data = data;
  newNode->next = NULL;
  if (isEmpty(q)) {
    q->front = newNode;
    q->rear = newNode;
  } else {
    q->rear->next = newNode;
    q->rear = newNode;
  }
  return 0;
}

int dequeue(Queue *q, int *data) {
  if (isEmpty(q)) {
    return -1;
  }
  Node *f = q->front;
  *data = f->data;
  q->front = f->next;
  if (q->front == NULL) {
    q->rear = NULL;
  }
  free(f);
  return 0;
}

void freeQueue(Queue *q) {
  int tmp;
  while (dequeue(q, &tmp) == 0) {
  }
}

void bfs(Queue *q, int start) {
  visited[start] = 1;
  enqueue(q, start);  // 初期頂点をキューに追加

  while (!isEmpty(q)) {
    int v;
    dequeue(q, &v);  // キューの先頭vを訪問
    printf("%d ", v);
    for (int i = 0; i < N; i++) {
      if (adj[v][i] && !visited[i]) {
        visited[i] = 1;  // vに隣接する未訪問頂点
        enqueue(q, i);   // キューに追加して次のvの隣接頂点へ
      }
    }
  }
}

int main(void) {
  // グラフの定義
  adj[0][1] = adj[1][0] = 1;
  adj[0][2] = adj[2][0] = 1;
  adj[1][3] = adj[3][1] = 1;
  adj[1][4] = adj[4][1] = 1;
  adj[2][4] = adj[4][2] = 1;

  Queue q;
  initQueue(&q);
  bfs(&q, 0);
  freeQueue(&q);
  return 0;
}
